Matrix-Matrix Products

(Source: Several Simple Real-world Applications of Linear Algebra Tools)

import numpy as np
A = np.array([[6, 5, 3, 1],
              [3, 6, 2, 2],
              [3, 4, 3, 1]])
B = np.array([[1.5, 1.0],
              [2.0, 2.5],
              [5.0, 4.5],
              [16.0, 17.0]])
A @ B
array([[50. , 49. ],
       [58.5, 61. ],
       [43.5, 43.5]])
